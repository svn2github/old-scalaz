package scalaz

sealed trait Alpha {
  val toChar: Char

  def toUpperChar = toChar.toUpper

  override def equals(o: Any) = o != null && o.isInstanceOf[Alpha] && o.asInstanceOf[Alpha].toChar == toChar

  override def hashCode = toChar.hashCode

  override def toString = toChar.toString
}
